private final List<List<Integer>> res = new ArrayList<>();

private void doSearchOpt(int[] nums, int start, int target, List<Integer> container) {

    if (container.size() == 1) {
        // åªè¦å°†é—®é¢˜é€’å½’è‡³ä¸‰æ•°ä¹‹å’Œé—®é¢˜ï¼Œå°±è¾¾åˆ°äº†ç®—æ³•çš„é€’å½’åŸº
        // å› ä¸ºä¸‰æ•°ä¹‹å’Œé—®é¢˜æœ‰è§£
        for (int i = start; i < nums.length; ++i) {
            if (i > start && nums[i] == nums[i - 1]) continue; // å»é‡
            int j = i + 1, k = nums.length - 1;
            while (j < k) {
                int sum = nums[i] + nums[j] + nums[k];
                if (sum > target){
                    k--;
                } else if (sum < target) {
                    j++;
                }else {
                    res.add(List.of(container.get(0), nums[i], nums[j], nums[k]));
                    // å»é‡
                    while (j < k && nums[j] == nums[j + 1]) j ++;
                    while (j < k && nums[k] == nums[k - 1]) k --;
                    j++; k--;
                }
            }

        }
        return;
    }

    for (int i = start; i < nums.length; ++i) {
        container.add(nums[i]);
        doSearchOpt(nums, i + 1, target - nums[i], container);
        container.remove(container.size() - 1); // å›æº¯
        while (i < nums.length - 1 && nums[i] == nums[i + 1]) i++; // å»é‡
    }
}


public List<List<Integer>> fourSum(int[] nums, int target) {
    if (nums.length < 4) return res;
    Arrays.sort(nums);
    doSearchOpt(nums, 0, target, new ArrayList<>());
    return res;
}
